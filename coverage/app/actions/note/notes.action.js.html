export const getNotes = () => dispatch => {
    return axios.get('/api/notes/')
        .then((response) => {
            dispatch({
                type: GET_NOTES,
                payload: response.data
            });
        })
        .catch(() => {
            dispatch(showFailModal({
                title: 'Error - could not retrieve notes',
                body: 'Note lookup failed, please try again later.'
            }))
        })
};
